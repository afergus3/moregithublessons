pipeline {
    agent any

    environment {
        DOCKERHUB_CREDENTIALS = credentials('dockerhub-creds') // Jenkins credentials ID for Docker Hub
        IMAGE_NAME = 'afergus3/moregithublessons'
        IMAGE_TAG = "latest"
    }

    stages {
        stage('Checkout') {
            steps {
                // Clone your Git repo containing Dockerfile and index.html
                git branch: 'main', url: 'https://github.com/afergus3/moregithublessons.git'
            }
        }

        stage('Docker Login') {
            steps {
                script {
                    // Login to Docker Hub using stored credentials
                    sh "echo $DOCKERHUB_CREDENTIALS_PSW | docker login -u $DOCKERHUB_CREDENTIALS_USR --password-stdin"
                }
            }
        }

        stage('Build Docker Image') {
            steps {
                script {
                    // Build the Docker image using your Dockerfile
                    dockerImage = docker.build("${IMAGE_NAME}:${IMAGE_TAG}")
                }
            }
        }

        stage('Push Docker Image') {
            steps {
                script {
                    // Push the image to Docker Hub
                    docker.withRegistry('', 'dockerhub-creds') {
                        dockerImage.push()
                    }
                }
            }
        }
    }

    post {
        success {
            echo "Docker image ${IMAGE_NAME}:${IMAGE_TAG} built and pushed successfully!"
        }
        failure {
            echo "Build or push failed."
        }
    }
}
